```python
from sklearn.metrics import mean_squared_error
from math import sqrt
```


```python
def main():
    #print("inside main method")
    
    y_actual = [1, 6, 7]
    y_predicted = [1, 6, 8]
    
    rms = sqrt(mean_squared_error(y_actual, y_predicted))
    
    print(rms)    
```


```python
if __name__ == '__main__':
    main()
```
