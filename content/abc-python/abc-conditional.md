# Python program to calculate the area of a triangle.
x = 4
y = 5
z = 6
# Calculating the semi-perimeter.
s = (x + y + z) / 2
# Calculating the area of triangle.
area = (s * (s -x) * (s -y) * (s - z)) ** 0.5
print('Area of triangle = %0.2f'%area)
