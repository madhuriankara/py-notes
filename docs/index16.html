def nested_list():
    lst = [[1, 2, 3], [4, 5], [6]]
    a = tuple(i for sublist in lst for i in sublist) 
    return a

def get_not_common_elements():
    set1 = {1, 2, 3, 4}
    set2 = {3, 4, 5, 6}
    not_common = (set1 - set2) | (set2 - set1)
    return not_common
