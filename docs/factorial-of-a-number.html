def factorial(n):
    """
    Calculate the factorial of a number.

    Parameters:
    n (int): The number to calculate the factorial for.

    Returns:
    int: The factorial of the given number.
    """

    # TODO: Implement the factorial calculation here
    if n ==0 or n ==1:
        return 1

    result = 1
    for i in range(2,n+1):
        result *= i
    return result
