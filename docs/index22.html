Takes two variables. Swaps their values without using a third variable. Prints the swapped values.

def swap_two_variables():
    a=10
    b=20
    a,b = b,a
    result = (a,b)
    return f'The {result} is now swapped'

swap_two_variables()

'The (20, 10) is now swapped'
