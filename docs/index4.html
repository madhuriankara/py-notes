import hashlib
import os

def get_file_hash(file_path):
    """Generate a hash for the contents of a file."""
    hash_sha256 = hashlib.sha256()
    with open(file_path, 'rb') as f:
        while chunk := f.read(4096):
            hash_sha256.update(chunk)
    return hash_sha256.hexdigest()

    if n == 0: return 0
