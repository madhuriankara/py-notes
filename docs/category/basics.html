from datetime import datetime

def get_age(d):
    d1 = datetime.now()
    months = (d1.year - d.year) * 12 + d1.month - d.month

    year = int(months / 12)
    return year

age = get_age(datetime(1991, 1, 1))

age

def print_everything(*args):
        for count, thing in enumerate(args):
            print( '{0}. {1}'.format(count, thing))

print_everything('apple', 'banana', 'cabbage', 'spinach')

import webcolors

def is_valid_hex_color(color):
    try:
        webcolors.hex_to_rgb(color)
        return True
    except Exception as e :
        return False

# Examples
print(is_valid_hex_color("#FFF"))  # True
print(is_valid_hex_color("#123ABC"))  # True
print(is_valid_hex_color("#123ABCG"))  # False
