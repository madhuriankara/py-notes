from pypika import Query, Table

# Define a table
users = Table("users")

# Build an INSERT query
query = Query.into(users).columns("id", "name", "age").insert(1, "John Doe", 28)

print(query)  # Output: INSERT INTO "users" ("id","name","age") VALUES (1,'John Doe',28)
