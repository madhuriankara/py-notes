def is_prime(num):
    """
    Check if a number is prime.

    Parameters:
    num (int): The number to check for primality.

    Returns:
    bool: True if the number is prime, False otherwise.
    """
    for n in range( 2, int(num**0.5)+1):   
        if (num % n) == 0:
            return False
        return True
